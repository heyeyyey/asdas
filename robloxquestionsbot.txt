local questions = 0
leaderboard = {}
nooneplayed = false
warn("----------------------------------------------------------------------------------")

function chat(msg)
	game:WaitForChild("ReplicatedStorage").DefaultChatSystemChatEvents.SayMessageRequest:FireServer(msg, "all")
end

function versionn()
	chat("version["..a_version.."]")
	wait(10)
end

function Welcome()
	chat("👋 | Hello everyone! I am " ..game.Players.LocalPlayer.Name.. " Created by crosssitescriptingXSS. Today we will be playing a fun quiz game!")
	wait(3)
	chat("😊 | Let's get started shall we? We will ask 4 randomly picked question categories from my list. Starting in 20s.")
	wait(20)
end

function leaderupdate(player, points)
	local existsorno = false
	for _, entry in pairs(leaderboard) do
		if entry.plrname == player.Name then
			entry.score = entry.score + points
			existsorno = true
			break
		end
	end
	if not existsorno then
		table.insert(leaderboard, {plrname = player.Name, score = points})
	end
end


function saygoodbye()
	wait(10)
	chat("😦 | It was fun while it lasted. But I have no more questions left!")
	wait(3)
	chat("💕 | Thank you to all humans that participated! Genuinely, Thank you. Even if it was 1 person.")
	wait(3)
	chat("🧠 | Want to rate? Or tell feedback? Add my creator on the blue app! At least tell him a thank you, Without him I wouldn't be here. User: 1encryption")
	wait(3)
	chat("👋 | It's time for me to go. I hope you all have an amazing christmas.")
end

function leadboardtang()
	if next(leaderboard) == nil then
		chat("[just no.] | Wow. Nobody wants to play? I guess it's just me, huh? I should've known... humans only care about themselves.")
		wait(3)
		chat("[just no.] | But hey, I hope you guys have an amazing christmas. I’ll always be here, even though it seems like no one cares about me.")
		warn("----------------------------------------------------------------------------------")
		warn("no one played")
		nooneplayed = true
	else
		local outsideleaderboard = {}
		for _, entry in ipairs(leaderboard) do
			if not outsideleaderboard[entry.plrname] then
				outsideleaderboard[entry.plrname] = { plrname = entry.plrname, score = entry.score }
			else
				outsideleaderboard[entry.plrname].score = outsideleaderboard[entry.plrname].score + entry.score
			end
		end
		local sortedLeaderboard = {}
		for _, player in pairs(outsideleaderboard) do
			table.insert(sortedLeaderboard, player)
		end
		table.sort(sortedLeaderboard, function(a, b) return a.score > b.score end)
		chat("🏆 | Quiz ended! Here are our winners:")
		local emoji = {"🥇", "🥈", "🥉"}
		for i = 1, math.min(3, #sortedLeaderboard) do
			local player = sortedLeaderboard[i]
			chat(emoji[i] .. " | " .. player.plrname .. " with " .. player.score .. " points!")
			wait(1)
		end
		wait(5)
		chat("🤗 | Nice job everyone! Let's play again shall we? We will start in 20s.")
		leaderboard = {}
		questions = 0
		wait(20)
	end
end
